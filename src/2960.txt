//에라토스테네스의 체 //

#include<stdio.h>

int main(void)
{
	int N, K, number;
	int num[1001] = {0, };
	int ct=0;
	scanf("%d %d", &N, &K);
	for(int i=2; i<=N; i++)
	{
		if(num[i] == 1)
			continue;
		for(int j=i; j<=N; j += i)
		{
			if(num[j] == 1)
				continue;
			num[j] = 1;
			ct++;
			if(ct==K)
			{
				number = j;
				break;
			}
		}
	}
	printf("%d\n", number);
	return 0;
}

#include <stdio.h>

int main(){
	int N,K,arr[1001] = {0,};
	int i,j,count=0;
	scanf("%d %d", &N, &K); 
	for(i=2; i<=N; i++){
		for(j=1; j*i<=N; j++){
			if(arr[j*i] == 0){
				count++;
				arr[j*i] = 1;
			}
			if(count == K){
				printf("%d", i*j); 
				goto EXIT;
			}
		}
	}
EXIT:
	return 0;
}

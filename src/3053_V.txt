//유클리드 기하학과 택시 기하학에 대한 문제


#include <stdio.h>
#include <math.h>
#define PI 3.14159265358979
int main(void)
{
	int R; 
	double Euclid, Taxi;				//double형으로 맞춰줌으로써 정밀도를 높이는 것이 핵심이었다.
	scanf("%d", &R);					//float형으로 변수 선언 시 오차범위 때문에 틀린다.
	Euclid = PI*R*R;
	Taxi = 2.0*R*R;
	printf("%f\n%f", Euclid, Taxi);
	
	return 0;
}

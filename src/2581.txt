//소수를 찾는 문제 2

#include<stdio.h>

int isPrime(int n)
{
	int counter = 0;
	for(int i=1; i<=n ;i++)
	{
		if(n%i == 0)
			counter++;
	}
	return (counter == 2);
}

int main(void)
{
	int A, B, least;
	int sum = 0; int ct = 0;
	scanf("%d\n%d", &A, &B);
	
	for(int x=A; x<=B; x++)
	{
		if(isPrime(x))
		{
			if(ct == 0)
				least = x;
			ct = 1;
			if(ct == 1)
				sum += x;
		}
	}
	if(sum == 0)
		printf("%d\n", -1);
	else
		printf("%d\n%d\n", sum, least);
}

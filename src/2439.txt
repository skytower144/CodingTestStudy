#include <stdio.h>

int main()
{
	int starNumber, z;
	int k = 1;
	
	scanf("%d", &starNumber);
	for(int x=0; x<starNumber; x++)
	{
		for(z = k; z<starNumber; z++)
		{
			printf(" ");
		}
		for(int y=0; y<=x; y++)
		{
			printf("*");
		}
		printf("\n");
		k++;
	}
	
	return 0;
}

#include <stdio.h>

int main(void)
{
	int n;
	int cnt;

	scanf("%d", &n);
	cnt = 1;
	while (cnt <= n)
	{
		for (int i = 0; i < n - cnt; i++)
			putchar(' ');
		for (int i = 0; i < cnt; i++)
			putchar('*');
		putchar('\n');
		++cnt;
	}
	return 0;
}

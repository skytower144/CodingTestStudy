//분수의 순서에서 규칙을 찾는 문제

#include<stdio.h>
#include<math.h>

int main(void)
{	
	int a,b,input, cut, even;
	a=1; b=1;
	scanf("%d", &input);
	cut = (pow(a,2) - a + 2) / 2;
	while(input>=cut)
	{
		if(input==cut)
		{
			if(a % 2 == 0)
			{
				printf("1/%d\n", a);
				return 0;
			}
			else
			{
				printf("%d/1\n", a);
				return 0;
			}
		}
		else
		{
			a++;
			cut = (pow(a,2) - a + 2) / 2;
		}		
	}
	a--;
	cut = (pow(a,2) - a + 2) / 2;
	even = a;
	while(cut<input)
	{
		a--;
		b++;
		cut++;
	}
	(even % 2 != 0) ? printf("%d/%d\n", a,b) : printf("%d/%d\n", b,a);
}

#include <stdio.h>

int main(void){
   int n,i;
   scanf("%d", &n);
   for(i=0;i<n;i++){
      if((i*(i+1))/2 >= n){
         break;
      }
   }
   n-=(i*(i-1)/2);
   if(i%2){
      printf("%d/%d", i-n+1, n);
   }
   else{
      printf("%d/%d",n,i-n+1);
   }
   return 0;
}

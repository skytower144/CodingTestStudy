// 소수 찾기

#include<stdio.h>
int isPrime(int n)
{
	int counter = 0;
	for(int i=1; i<=n ;i++)
	{
		if(n%i == 0)
			counter++;
	}
	return (counter == 2);
}

int main(void)
{
	int testC;
	int number = 0;
	int num[101] = {0, };
	scanf("%d", &testC);
	for (int i=0; i<testC; i++)
	{
		scanf("%d", &num[i]);				// &을 붙이는 걸 까먹지 말자.
		if(isPrime(num[i]))
			number++;
	}
	printf("%d\n", number);
	
	return 0;
}

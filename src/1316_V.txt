//그룹 단어 체커_조건에 맞는 문자열을 찾는 문제
#include<stdio.h>
#include<string.h>

int main()
{
	char words[101];
	int input;
	int counter = 0;
	
	scanf("%d", &input);
	for(int x=0; x<input; x++)
	{
		scanf("%s", words);
		int len = strlen(words);
		
		int flag[26] = {0,};						//매번 이 flag를 비우는 걸 까먹지 말자.
		int combo = 1;
		for(int y=0; y<len; y++)
		{
			if(flag[words[y]-'a']>0)
			{
				if(combo == 1)
					goto SKIP;
			}		
			flag[words[y]-'a']++;
			if(words[y] == words[y+1])
				combo++;
			else
				combo = 1;
		}
		counter++;
		SKIP:
			continue;
	}
	printf("%d\n", counter);
}

#include<stdio.h>
#include<string.h>

int self_num(char *str)
{
    int len=strlen(str);

    for(int i=0;str[i];i++)
        for(int j=0;j<len;j++)
            if(str[i]==str[j]&& j-i>1 && (str[j]!=str[j-1]))
                return 0;
    return 1;
}

int main(void)
{
    int num;
    int sum=0;
    char str[101];
    scanf("%d",&num);

    for(int i=0; i<num; i++)
    {
        scanf("%s",str);
        sum += self_num(str);
    }
    printf("%d\n",sum);
}

#include <stdio.h>

int isHansu(int n)
{
	int a1, a2, a3;
	int dif_1, dif_2;
	if(n<100)
	{
		return 1;
	}
	else if(n>=100 && n<1000)
	{
		a3 = n%10;
		a2 = (n/10)%10; 
		a1 = n /100;
		dif_1 = a1-a2;
		dif_2 = a2-a3;
		if(dif_1 == dif_2)
		{
			return 1;
		}
		else
		{
			return 0;
		}
	}
	else if(n == 1000)
	{
		return 0;
	}
}

int main()
{
	
	int number;
	int ct = 0;
	scanf("%d", &number);
	for(int x=1; x<=number; x++)
	{
		if(isHansu(x))
		{
			ct++;
		}
	}
	printf("%d\n", ct);
	
	
	return 0;
}

//팩토리얼//재귀 함수의 기본//

#include <stdio.h>

int factorial(int a)
{	
	int A;
	if(a==1 || a==0)						// 0팩토리얼 또한 1이라는 것이 핵심!
		return 1;
	else
		A = a * factorial(a-1);
	return A;
}

int main(void)
{
	int N; int result;
	scanf("%d", &N);
	result = factorial(N);
	printf("%d\n", result);
	
	return 0;
}

/////////////////////////////////////////////////////////////////////////////
다른 풀이)

int factorial(int N)
{
	if (N == 1 || N == 0)	return 1;
	else					return N * factorial(N - 1);
}


/////////////////////////////////////////////////////////////////////////////

int fac(int n) 
{
	if(n > 1)
		return n * fac(n - 1);
	else if(n == 0)
		return 1; 
	else
		return n; 
}

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
int main(void)
{
	char input[10001];
	int ct=0;
	int i; int H; 
	scanf("%s", input);
	H = atoi(input);
	
	for(i=1; ct < H; i++)
	{
		char buf[10001];
		int min_ct = 0;				//i가 갱신될 때마다 min_ct를 매번 초기화시켜줘야 한다!
		sprintf(buf,"%d",i);
		int LEN = strlen(buf);
		for(int k=0; k<LEN; k++)
		{
			if(buf[k] == '6')
				min_ct++;
			else
				min_ct = 0;
			if(min_ct == 3)
			{
				ct++;
				break;
			}
		}
	}
	printf("%d\n", i-1);
	
	return 0;
}

#include<stdio.h>

int main(void)
{	
	int A,B,V, times;
	scanf("%d %d %d", &A, &B, &V);
	int dif = A-B;
	int goal = V-A;
	if(A>=V)
	{
		printf("1\n");
		return 0;
	}
	times = goal / dif;
	if(goal % dif == 0)
		printf("%d\n", times + 1);
	else	
		printf("%d\n", times + 2);
	return 0;
}

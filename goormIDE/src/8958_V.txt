//OX 퀴즈의 결과를 일차원 배열로 입력받아 점수를 계산하는 문제
#include <stdio.h>
#include <string.h>

int main()
{
	int testNumber, counter, sum;
	counter = sum = 0;
	char s[80];
	int list[80];
	scanf("%d", &testNumber);
	
	for(int x=0; x<testNumber; x++)
	{
		
		scanf("%s", s);
		for(int y=0; y<strlen(s); y++)
		{
			if(s[y] == 'O')
			{
				counter++;
				list[y] = counter;
			}
			else
			{
				counter = 0;
				list[y] = counter;
			}
			sum += list[y];
		}
		printf("%d\n", sum);
		sum = 0;                        //2번째 테스트케이스로 넘어가기 전 초기화가 중요!
		counter = 0;					//상동
	}
	
	return 0;
}

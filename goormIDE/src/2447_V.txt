#include <iostream>
using namespace std;

void Star(int i, int j, int N) {
	if (i % 3 == 1 && j % 3 == 1) {
		cout << " ";
		return;
	}
	else if (N == 1) {
		cout << "*";
		return;
	}
	else Star(i / 3, j / 3, N / 3);
}

int main() {
	double N;
	cin >> N;
	for (int i = 0; i < N; i++) {
		for (int j = 0; j < N; j++) {
			Star(i, j, N);
		}
		cout << "\n";
	}
	return 0;
}

#include <stdio.h>

void star(int a, int b, int n)
{
	if(a % 3 == 1 && b % 3 == 1)
		printf(" ");
	else if(n==1)
		printf("*");
	else
		star(a/3, b/3, n/3);
}

int main(void)
{
	int N;
	scanf("%d", &N);
	
	for(int y=0; y<N; y++)
	{
		for(int x=0; x<N; x++)
		{
			star(x,y,N);
			if(x==N-1)
				printf("\n");
		}
	}
	
	return 0;
	
}

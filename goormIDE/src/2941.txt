#include<stdio.h>
#include<string.h>

int main()
{
	char input[200];
	scanf("%s", input);
	int len = strlen(input);
	int counter = 0;
	
	for(int x=0; x<len; x++)
	{
		if(input[x] == 'c')
		{
			if(input[x+1] == '=' || input[x+1] == '-')
				x += 1;
		}
		else if(input[x] == 'd')
		{
			if(input[x+1] == 'z' && input[x+2] == '=')
				x += 2;
			else if(input[x+1] == '-')
				x += 1;
		}
		else if(input[x] == 'l')
		{
			if(input[x+1] == 'j')
				x += 1;
		}
		else if(input[x] == 'n')
		{
			if(input[x+1] == 'j')
				x += 1;
		}
		if(input[x] == 's')
		{
			if(input[x+1] == '=')
				x += 1;
		}
		if(input[x] == 'z')
		{
			if(input[x+1] == '=')
				x += 1;
		}
		counter++;
	}
	printf("%d\n", counter);
}

	{
		if (((str[i] == 'c') || (str[i] == 'd')) && (str[i+1] == '-'))

		if (((str[i] == 's') || (str[i] == 'z') || (str[i] == 'c')) && (str[i + 1] == '='))
			continue;

			continue;
		if (str[i] == 'd' && str[i + 1] == 'z' && str[i + 2] == '=')
			continue;
		count++;
	}

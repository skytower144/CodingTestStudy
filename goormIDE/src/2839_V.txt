#include<stdio.h>

int main(void)
{	// c = 5*a + 3*b
	int a, b, c;
	
	scanf("%d", &c);
	
	if((c % 5) % 3 == 0)
	{
		a = c/5;
		b = (c%5) / 3;
		printf("%d\n", a+b);
		return 0;
	}
	else
	{
		int k = (c/5);
		int j;
		while(k>0)
		{
			k--;
			if((c - 5*k) % 3 == 0)				// == 0을 붙여줘야 한다.
			{
				j = (c - 5*k) / 3;
				printf("%d\n", k+j);
				return 0;
			}
		}
		if(c % 3 == 0)
		{
			printf("%d\n", c/3);
			return 0;
		}	
	}
	printf("%d\n", -1);
}

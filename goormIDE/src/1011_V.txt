//Fly me to the Alpha Centauri_거리에 따른 장치 사용 횟수를 출력하는 문제

#include<stdio.h>
#include<math.h>

int main(void)
{
	int testC;
	long a, b;								//int형으로 하면 틀린다. long형으로 맞춰줘야함.
	scanf("%d", &testC);
	for(int t=0; t<testC; t++)
	{
		scanf("%ld %ld", &a, &b);
		long n, mid;
		n = (long)sqrt(b-a)+1;					//math 헤더파일을 통해 sqrt함수를 써야함. 시간초과 때문에.
		long distance = b-a;
		mid = ((n-1)*(n-1) + n*n)/2;
		
		if((n-1)*(n-1) == distance)
		{
			printf("%ld\n", 2*n-3);
		}
		else if(distance <= mid)
		{
			printf("%ld\n", 2*n-2);
		}
		else if(distance > mid)
		{
			printf("%ld\n", 2*n-1);
		}
	}
	return 0;
}

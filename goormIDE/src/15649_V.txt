#include <stdio.h>

int N, M;
int memory[10] ={0,};
int boo[10] = {0, };
int dfs(int ct)
{
	if(ct == M)
	{
		for(int x=0; x<M; x++)
		{
			printf("%d ", memory[x]);
		}
		printf("\n");
		return 0;
	}
	for(int i=1; i<=N; i++)
	{
		if(boo[i] != 1)
		{
			boo[i] = 1;
			memory[ct] = i;
			dfs(ct+1);
			boo[i] = 0;	
		}
	}
}

{
	scanf("%d %d", &N, &M);
	dfs(0);
	
	return 0;
}

#include <iostream>
#define MAX 9
using namespace std;

int n,m;
int arr[MAX] = {0,};
bool visited[MAX] = {0,};

void dfs(int cnt)
{
    if(cnt == m)
    {
        for(int i = 0; i < m; i++)
            cout << arr[i] << ' ';
        cout << '\n';
        return;
    }
    for(int i = 1; i <= n; i++)
    {
        if(!visited[i])
        {
            visited[i] = true;
            arr[cnt] = i;
            dfs(cnt+1);
            visited[i] = false;
        }
    }
}

int main() {
    cin >> n >> m;
    dfs(0);
}

//스타트와 링크// 백트래킹// 

#include <stdio.h>
#include <stdlib.h>
int N;
int n[21][21];
int flag[21];
int MIN = 100000;
int dif;
int sum1; int sum2;
int a1[21] = {0,};
int a2[21] = {0,};
int b=0;

int cal(int y, int x)
{
	return n[y-1][x-1];
}

void starlink(int ct)
{
	if(ct==N/2)
	{
		for(int i=1; i<=N; i++)
		{
			if(flag[i] == 0)
			{
				a2[b++] = i;
			}
		}
		b=0;
		for(int y=0; y<N/2; y++)
		{
			for(int x=0; x<N/2; x++)
			{
				if(y==x)
					continue;
				else
				{
					sum1 += cal(a1[y],a1[x]);
					sum2 += cal(a2[y],a2[x]);
				}
			}
		}
		dif = abs(sum1 - sum2);
		if(dif < MIN)
			MIN = dif;
		sum1=0; sum2=0;
	}
	for(int i=1; i<=N; i++)
	{
		if(ct==0)
		{
			flag[i] = 1;
			a1[ct] = i;
			starlink(ct+1);
			flag[i] = 0;
		}
		else if(i > a1[ct-1])
		{
			flag[i] = 1;
			a1[ct] = i; 
			starlink(ct+1);
			flag[i] = 0;
		}
		if(i==1 && ct==0)
			break;
	}
}

int main()
{
	scanf("%d", &N);
	for(int i=0; i<N; i++)
	{
		for(int k=0; k<N; k++)
		{
			scanf("%d", &n[i][k]);
		}
	}
	starlink(0);
	printf("%d\n", MIN);
	return 0;
}

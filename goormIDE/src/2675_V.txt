//각 문자를 반복하여 출력하는 문제
#include<stdio.h>
#include<string.h>

int main()
{
	int testC;
	scanf("%d", &testC);
	for(int x=0; x<testC; x++)
	{
		int rpt, counter;
		int b = 0;
		char qr[21] = "";					//20으로 설정하면 틀림. // 문자열의 끝은 항상'\0'이 들어가기 											//				  때문에 배열크기를 1크게 잡아주셔야 합니다.
		char bigQr[168] = "";				//초기화 시켜주지않으면 잔지꺼기가 계속 남음. 왜그런지 모르겠음.
		scanf("%d %s", &rpt, &qr);
		for(int a=0; a<strlen(qr); a++)
		{
			counter = 0;
			while(counter<rpt)
			{
				bigQr[b] = qr[a];
				b++;
				counter++;
			}
		}
		printf("%s\n", bigQr);
	}			   
}

#include <stdio.h>
#include <string.h>

main(){
	char arr[21];
	char coppy_arr[200];
	int test,re,k,i,j;
	scanf("%d",&test);
	for(i=0;i<test;i++){
		scanf("%d",&re);
		scanf("%s",arr);
		for(j=0;j<=strlen(arr);j++){
			for(k=0;k<re;k++){
				coppy_arr[re*j+k]=arr[j];               // (re 곱하기 j) + k
			}											// j가 곱하기 카운터 느낌
		}
		printf("%s\n",coppy_arr);	
	}	
}

//원래 배열을 반복해서 출력하는 방식

#include<stdio.h>
#include<string.h>

int main() {
    int t, r;
    char s[21];
    scanf("%d", &t);
    
    for(int i=0; i<t; i++){
        scanf("%d %s", &r, s);
        
        for(int j=0; j<strlen(s); j++){
            for(int k=0; k<r; k++){
                printf("%c", s[j]);
            }
        }
        printf("\n");
    }
}

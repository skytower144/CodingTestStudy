//수 정렬하기//

//삽입 정렬_my version

#include <stdio.h>

int main(void)
{
	int N;
	int num[1001] = {0, };
	scanf("%d", &N);
	for(int u=0; u<N; u++)
	{
		scanf("%d", &num[u]);
	}
	
	for(int i=1; i<N; i++)
	{
		int answer = 0; int temp = 0; int light = 0;
		for(int j=i-1; j>=0; j--)
		{
			if(num[i] < num[j])
			{
				answer = j;
				light = 1;
			}
		}
		if(light)
		{
			temp = num[i];
			for(int k = i-1; k>=answer; k--)
			{
				num[k+1] = num[k];
			}
			num[answer] = temp;
		}
	}
	for(int x=0; x<N; x++)
	{
		printf("%d\n", num[x]);
	}
	return 0;
}

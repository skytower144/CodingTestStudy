//01타일 // 점화식의 값을 특정 상수로 나눈 나머지를 구하는 문제//

#include <stdio.h>
int f[1000002] = {0, 1, 2, 0};
int fibo(int n)
{
	for(int i=3; i<=n; i++)
	{
		f[i] = (f[i-1] + f[i-2])%15746;		//핵심!
	}
	return f[n];
}
int main()
{
	int N;
	scanf("%d", &N);
	printf("%d\n", fibo(N));
	
	return 0;
}

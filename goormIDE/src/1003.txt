//동적 계획법1// 피보나치 함수

#include <stdio.h>

int cal0(int n)
{
	int result = 0;
	int preSum = 1;
	int sum = 0;
	if(n==0)
		return 1;
	else if(n==1)
		return 0;
	else
	{
		for(int i=2; i<=n; i++)
		{
			result = preSum + sum;
			preSum = sum;
			sum = result;
		}
		return result;
	}
}
int cal1(int n)
{
	int result = 0;
	int preSum = 0;
	int sum = 1;
	if(n==0)
		return 0;
	else if(n==1)
		return 1;
	else
	{
		for(int i=2; i<=n; i++)
		{
			result = preSum + sum;
			preSum = sum;
			sum = result;
		}
		return result;
	}
}

int main()
{
	int T; int N;
	scanf("%d", &T);
	for(int i=0; i<T; i++)
	{
		scanf("%d", &N);
		printf("%d %d\n", cal0(N), cal1(N));
	}
	return 0;
}

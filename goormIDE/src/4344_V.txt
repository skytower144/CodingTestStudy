// "평균" 구하기와 "평균을 넘는 점수의 비율" 구하기
#include <stdio.h>

int main()
{
	
	int testNumber, studentNumber;
	float sum, score, average, counter, percentage;
	sum = 0;
	int classRoom[1000] = {0, };
	scanf("%d", &testNumber);
	
	for(int x=0; x<testNumber; x++)
	{
		scanf("%d", &studentNumber);
		
		for(int y=0; y<studentNumber; y++)
		{
			scanf("%f", &score);
			sum += score;
			classRoom[y] = score;
		}
		average = sum / studentNumber;
		for(int z=0; z<studentNumber; z++)
		{
			if(average<classRoom[z])
			{
				counter++;
			}
		}
		percentage = counter / studentNumber * 100;
		printf("%.3f%\n", percentage);
		
		sum = 0;
		counter = 0;
		for(int v=0; v<studentNumber; v++)
		{
			classRoom[v] = 0;
		}
	}
	
	return 0;
}

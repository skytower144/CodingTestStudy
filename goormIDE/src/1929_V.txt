// 에라토스테네스의 체 // 소수 구하기

#include<stdio.h>
#include<math.h>
int isPrime(int n)
{
	int counter = 0;
	for(int i=1; i<=n ;i++)
	{
		if(n%i == 0)
			counter++;
	}
	return (counter == 2);
}
int main(void)
{
	int n1, n2, A;
	int number[1000001] = { 1, 1, 0, };  			// 숫자 0과 1을 제외시키는 작업을 잊지 말자.
	scanf("%d %d", &n1, &n2);
	
	for(A = 2; A<=n2; A++)					// 2부터 시작하자.
	{
		if(pow(A,2) > n2)				// >= 해도 상관이 없었다. 왠지는 모르겠음. 머리가 안 돌아감.
		{								// 시간을 줄이기 위해 제곱의 형태로 비교. 에라토스테네스의 체 위키 참고.
			break;						// 굳이 비교 안해도 되는 숫자들을 미리 배제시킴.
		}
		if(isPrime(A))
		{
			for(int x=n1; x<=n2; x++)
			{
				if(x == A)
					number[x] = 0;
				else if(x%A == 0)
				{
					number[x] = 1;
				}
			}	
		}
		else
			continue;
	}
	for(int y=n1; y<=n2; y++)
	{
		if(number[y] == 0)
		{
			printf("%d\n", y);
		}
	}
	return 0;
}
